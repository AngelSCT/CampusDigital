//COMANDOS PARA CORRER LA APLICACION

//CLONAR EL REPOSITORIO
git clone https://github.com/AngelSCT/CampusDigital

//ENTRAR A LA CARPETA DEL PROYECTO LLAMADA "campus-digital-app" (OJO AQUI ES DONDE SE EJECUTARAN TODOS LOS COMANDOS DE INSTALACION DE DEPENDENCIAS, ETC):
1) EJECUTAR LA TERMINAL EN VS CODE
2) USANDO EL COMANDO "cd" DESPLAZARNOS A LA CARPETA "campus-digital-app"
3) VERIFICAR QUE ESTEMOS EN ESA CARPETA

//UNA VEZ DENTRO DE LA CARPETA DEL PROYECTO ("campus-digital-app") EJECUTAMOS LOS SIGUIENTES COMANDOS:

-SI USAS LINUX:

rm -rf node_modules
rm package-lock.json
composer install
npm install

-SI USAS WINDOWS:

Remove-Item node_modules -Recurse -Force
Remove-Item package-lock.json -Force
composer install
npm install

//AHORA CREAR EL ARCHIVO .ENV 
1) DENTRO DE LA CARPETA DEL PROYECTO "campus-digital-app" CREAR UN ARCHIVO LLAMADO ".env" (SIN COMILLAS)
2) DENTRO DE ESE ARCHIVO .env PEGAR EL CONTENIDO DEL ARCHIVO "archivo-.env.txt" EL CUAL ESTA EN LA CARPETA DE "archivos-adicionales"
3) GUARDAR LOS CAMBIOS EN EL ARCHIVO .env (ctrl + s)
4) UNA VEZ ESTE EL ARCHVO .ENV EJECUTAR LOS SIGUIENTES COMANDOS:

-SI USAS LINUX:

php artisan key:generate
php artisan optimize:clear

-SI USAS WINDOWS:

php artisan key:generate
php artisan optimize:clear


//AHORA COMO SIGUIENTE PASO CREAREMOS EL ESQUEMA DE LA BASE DE DATOS PARA QUE FUNCIONEN CORRECTAMENTE LAS MIGRACIONES Y ADEMAS CREAREMOS EL USUARIO DE LA BASE DE DATOS

1) DENTRO DE POSTGRES EJECUTAR LOS SIGUIENTES COMANDOS:

CREATE USER campus_user WITH PASSWORD 'postgres';
CREATE DATABASE campus_digital OWNER campus_user;
GRANT ALL PRIVILEGES ON DATABASE campus_digital TO campus_user;


2) UNA VEZ CREANDO LA BASE DE DATOS EN POSTGRES Y EL USUARIO, PROCEDEMOS A EJECUTAR LOS SIGUIENTES COMANDOS EN VSCODE:

php artisan migrate
php artisan db:seed


//UNA VEZ QUE SE HAYAN CARGADO LAS MIGRACIONES Y LOS DATOS DE PRUEBA PODREMOS VER LA APLICACION FUNCIONANDO
//PARA ESTO USAREMOS 2 TERMINALES , AMBAS APUNTANDO A LA RUTA DE LA CARPETA DEL PROYECTO

1)TERMINAL 1 COMANDO(EJECUTAR LARAVEL):

php artisan serve

2)TERMINAL 2 COMANDO(EJECUTAR FRONTEND):

npm run dev


// UNA VEZ CORRIENDO LARAVEL + FRONTEND, PROCEDEMOS A ABRIR EL NAVEGADOR Y COLOCAR EN LA URL: localhost:8000 
// ESE SERA EL PUERTO DONDE SE EJECUTA LARAVEL O LA APLICACION
